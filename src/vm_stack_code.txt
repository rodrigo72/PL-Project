INCPOINTER:
	PUSHG 0 DUP 1 LOAD 0 PUSHI 1 ADD STORE 0 RETURN

DECPOINTER:
	PUSHG 0  DUP 1 LOAD 0 PUSHI 1 SUB STORE 0 RETURN

MYPUSH:
	PUSHG 0 LOAD 0
	PUSHG 1 SWAP PADD PUSHFP LOAD -1 STORE 0
	JUMP INCPOINTER
	RETURN

MYPOP:
	PUSHG 0 LOAD 0 PUSHI 1 SUB
	PUSHG 1 SWAP PADD LOAD 0
	JUMP DECPOINTER
	RETURN
